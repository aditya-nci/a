Customer Lifetime Value Analysis

WITH customer_metrics AS ( SELECT customer_id, COUNT(DISTINCT order_id) as total_orders, SUM(order_total) as lifetime_value, AVG(order_total) as avg_order_value, DATEDIFF(MAX(order_date), MIN(order_date)) as customer_lifespan, RANK() OVER (ORDER BY SUM(order_total) DESC) as value_rank FROM orders o JOIN customers c ON o.customer_id = c.id WHERE order_date >= DATE_SUB(CURDATE(), INTERVAL 2 YEAR) GROUP BY customer_id ), retention_analysis AS ( SELECT customer_id, LAG(order_date) OVER (PARTITION BY customer_id ORDER BY order_date) as prev_order, DATEDIFF(order_date, LAG(order_date) OVER (PARTITION BY customer_id ORDER BY order_date)) as days_between_orders FROM orders ) SELECT cm.customer_id, cm.lifetime_value, cm.total_orders, cm.avg_order_value, cm.value_rank, AVG(ra.days_between_orders) as avg_purchase_frequency FROM customer_metrics cm LEFT JOIN retention_analysis ra ON cm.customer_id = ra.customer_id WHERE cm.total_orders >= 3 GROUP BY cm.customer_id, cm.lifetime_value, cm.total_orders, cm.avg_order_value, cm.value_rank ORDER BY cm.lifetime_value DESC;

Revenue Trend Analysis with Moving Averages

SELECT DATE_FORMAT(order_date, '%Y-%m') as month, SUM(order_total) as monthly_revenue, AVG(SUM(order_total)) OVER ( ORDER BY DATE_FORMAT(order_date, '%Y-%m') ROWS BETWEEN 2 PRECEDING AND CURRENT ROW ) as three_month_moving_avg, LAG(SUM(order_total), 1) OVER (ORDER BY DATE_FORMAT(order_date, '%Y-%m')) as prev_month_revenue, ROUND( ((SUM(order_total) - LAG(SUM(order_total), 1) OVER (ORDER BY DATE_FORMAT(order_date, '%Y-%m'))) / LAG(SUM(order_total), 1) OVER (ORDER BY DATE_FORMAT(order_date, '%Y-%m'))) * 100, 2 ) as month_over_month_growth FROM orders WHERE order_date >= DATE_SUB(CURDATE(), INTERVAL 12 MONTH) GROUP BY DATE_FORMAT(order_date, '%Y-%m') ORDER BY month;
